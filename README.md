## Install

```bash
npm install --save base-service
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'base-service'
import 'base-service/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
